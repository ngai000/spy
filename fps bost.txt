-- Bi·∫øn tr·∫°ng th√°i
local isLagFixEnabled = false

-- Bi·∫øn ƒë·ªÉ t√≠nh FPS
local lastTime = tick()
local frameCount = 0

-- C·∫≠p nh·∫≠t FPS
game:GetService("RunService").RenderStepped:Connect(function()
    frameCount = frameCount + 1
    local currentTime = tick()
    if currentTime - lastTime >= 1 then
        local fps = frameCount / (currentTime - lastTime)
        TextLabel.Text = string.format("FPS: %.0f", fps)
        frameCount = 0
        lastTime = currentTime
    end
end)

-- Script gi·∫£m lag h·ª£p ph√°p trong Roblox
local Players = game:GetService("Players")
local Lighting = game:GetService("Lighting")

local function optimizeGraphics()
    Lighting.GlobalShadows = false
    Lighting.FogEnd = 100000
    Lighting.Brightness = 2
    Lighting.OutdoorAmbient = Color3.new(1, 1, 1)

    for _, part in pairs(workspace:GetDescendants()) do
        if part:IsA("BasePart") then
            part.Material = Enum.Material.SmoothPlastic
            part.CastShadow = false
        elseif part:IsA("ParticleEmitter") or part:IsA("Trail") then
            part.Enabled = false
        elseif part:IsA("Decal") or part:IsA("Texture") then
            part.Transparency = 0.5
        end
    end

    local player = Players.LocalPlayer
    if player.Character then
        for _, obj in pairs(player.Character:GetDescendants()) do
            if obj:IsA("ParticleEmitter") or obj:IsA("Trail") then
                obj.Enabled = false
            end
        end
    end
end

local function resetGraphics()
    Lighting.GlobalShadows = true
    Lighting.FogEnd = 1000
    Lighting.Brightness = 1
    Lighting.OutdoorAmbient = Color3.new(0.5, 0.5, 0.5)

    for _, part in pairs(workspace:GetDescendants()) do
        if part:IsA("BasePart") then
            part.Material = Enum.Material.Plastic
            part.CastShadow = true
        elseif part:IsA("ParticleEmitter") or part:IsA("Trail") then
            part.Enabled = true
        elseif part:IsA("Decal") or part:IsA("Texture") then
            part.Transparency = 0
        end
    end

    local player = Players.LocalPlayer
    if player.Character then
        for _, obj in pairs(player.Character:GetDescendants()) do
            if obj:IsA("ParticleEmitter") or obj:IsA("Trail") then
                obj.Enabled = true
            end
        end
    end
end

-- T·∫Øt/B·∫≠t fix lag khi nh·∫•n n√∫t
ToggleButton.MouseButton1Click:Connect(function()
    isLagFixEnabled = not isLagFixEnabled

    if isLagFixEnabled then
        optimizeGraphics()
        ToggleButton.Text = "Fix Lag ON"
    else
        resetGraphics()
        ToggleButton.Text = "Fix Lag OFF"
    end
end)
